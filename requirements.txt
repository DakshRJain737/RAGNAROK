# ─────────────────────────────────────────────────────────────────────────────
# requirements.txt — Redrob Candidate Ranking System
# Python 3.12 · CPU-only · No GPU · No network during ranking
#
# Install: pip install -r requirements.txt
#
# Pinning strategy:
#   - Exact pins (==) for ML/NLP packages — these have breaking changes
#     between minor versions and must be reproducible on judge's machine.
#   - Compatible-release pins (~=) for utilities — patch updates are safe.
#   - torch is CPU-only; the +cpu build avoids pulling the 2GB CUDA variant.
#
# Two install paths:
#   FULL (dev machine, pre-compute):
#       pip install -r requirements.txt
#   RANKING ONLY (judge sandbox, 5-min window):
#       pip install -r requirements.txt --no-deps  # indexes already built
# ─────────────────────────────────────────────────────────────────────────────


# ── Core ML / NLP ─────────────────────────────────────────────────────────────

# Bi-encoder (all-MiniLM-L6-v2) + cross-encoder (ms-marco-MiniLM-L-6-v2).
# 3.4.1 is the last stable release before the 4.x API restructure.
sentence-transformers==3.4.1

# PyTorch CPU-only build. Avoids the 2GB+ CUDA wheels.
# torch 2.5.1 is the stable LTS-aligned release for Python 3.12.
# Must install BEFORE sentence-transformers to avoid wheel conflicts.

# Hugging Face Transformers — sentence-transformers dependency.
# 4.57.6 is the last stable 4.x release; 5.x has breaking API changes.

# Tokenizers — required by transformers. Exact pin prevents ABI mismatches.

# ─────────────────────────────────────────────────────────────────────────────
# POST-INSTALL STEPS (run once after pip install)
# ─────────────────────────────────────────────────────────────────────────────
#
# 1. Download spaCy language model (required by pipeline/jd_parser.py):
#       python -m spacy download en_core_web_sm
#
# 2. Pre-download sentence-transformer models (required by precompute.py):
#       python -c "from sentence_transformers import SentenceTransformer; \
#                  SentenceTransformer('all-MiniLM-L6-v2'); \
#                  from sentence_transformers import CrossEncoder; \
#                  CrossEncoder('cross-encoder/ms-marco-MiniLM-L-6-v2')"
#
# Both models are cached in ~/.cache/huggingface/ after first download.
# During ranking (5-min window, no network), they load from cache.
# Total cache size: ~22MB (bi-encoder) + ~80MB (cross-encoder) = ~102MB.
#
# 3. Run pre-computation (outside 5-min window):
#       python precompute.py --candidates data/candidates.jsonl.gz
#
# 4. Verify everything works on sample:
#       python rank.py --candidates data/sample_candidates.json --out output/test.csv
#       python validate_submission.py output/test.csv
# ─────────────────────────────────────────────────────────────────────────────
